Then deploy ingress on the cluster 

kubectl apply -f deploy/kubernetes/ingress/mushop-dev.yaml
apiVersion: extensions/v1beta1
kind: Ingress
metadata:
  name: mushop-dev
  annotations:
    kubernetes.io/ingress.class: nginx
spec:
  rules:
  - http:
      paths:
      - path: /
        backend:
          serviceName: edge
          servicePort: 80
kubectl get ingress mushop-dev
